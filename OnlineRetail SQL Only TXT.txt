WITH cohorts AS (
  SELECT
    Customer_ID,
    cast(MIN(DATEPART(year, InvoiceDate))as varchar) +' '+ cast(MIN(Datepart(month, invoicedate))as varchar) AS CohortMonth
  FROM onRet
  GROUP BY Customer_ID
)
SELECT
  CohortMonth,
  cast(datepart(year,invoiceDate)as varchar)+' '+cast(DATEPART(month, InvoiceDate)as varchar) AS PurchaseMonth,
  COUNT(DISTINCT a.Customer_ID) AS ActiveCustomers
FROM onRet a
JOIN cohorts b on a.Customer_ID = b.Customer_ID
GROUP BY CohortMonth, cast(datepart(year,invoiceDate)as varchar)+' '+cast(DATEPART(month, InvoiceDate)as varchar)
order by CohortMonth, PurchaseMonth;

-- Next I will calculate what percentage of revenue that comes from the top 20% of customers by revenue. I will use ntile(5), a view and a subquery



create view TRCustomerRevenue as
(
    select
        ntile(5) over(order by totrev desc) trank,
        *
        from
        (
            select
                customer_ID,
                sum(totalrevenue) totrev
            from onret
            where customer_ID != 'guest'
            group by Customer_ID 
        )t
)

--I'll use these to check my query
select sum(totrev) from TRCustomerRevenue where trank =1;
select sum(totrev) from TRCustomerRevenue;


--Query below calculates the answer (as 0.78)
select top 1 cast((select sum(totrev) from TRCustomerRevenue where trank =1) as float)/cast((select sum(totrev) from TRCustomerRevenue)as float) from TRCustomerRevenue;

--Next I will create a customer lifetime revenue distribution using case on the same view created above

select 
*,
case 
    when totrev between 0 and 999 then '$0-$1000'
    when totrev between 1000 and 2999 then '$1000-$3000'
    when totrev between 3000 and 4999 then '$3000-$5000'
    when totrev between 5000 and 6999 then '$5000-$7000'
    when totrev between 7000 and 8999 then '$7000-$9000'
    when totrev between 9000 and 9999 then '$9000-$10000'
    else '$10000+'
END revenueBand
from TRCustomerRevenue;


-- I will now do a 90 day churn detection analysis relative to the last purchase date in the dataset
-- The final table will return all customers, the gap between their last purchase and the latest date on the dataset and a flag if they haven't purchase for over 90 days


select
*,
case
    when orderGap>90 then 'churn detected'
    else 'ok'
    end churn
from
(
    select
        customer_ID,
        DATEDIFF(day, max(invoicedate), (select max(invoicedate) mx from onret)) orderGap
    from onret
    group by Customer_ID
)t
order by ordergap desc;

-- Below I will calculate the average number of days between orders per customer. I have also included the total number of orders per customer
-- To give more meaning to those averages.







select
    customer_ID,
    avg(daysSinceLastOrder) avgDaysBetweenOrders,
    count(distinct invoice) noOfOrders

    select
        *,

)t

with CountryAnalysis as
(
    select 
        country,

        sum(quantity*price) tr

    country,
    count(distinct invoice) noOfOrders,
    avg(tr) avgRevCountry,
    stdev(tr) stDevCountry,
    stdev(tr)/avg(tr) CofV

group by country;
